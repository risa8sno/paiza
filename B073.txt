import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        String[] line = sc.nextLine().split(" "); 
        int tree = Integer.parseInt(line[0]);           // 1行目：木の本数
        int light = Integer.parseInt(line[1]);          // 1行目：基準の個数

        String[] nowLight = sc.nextLine().split(" ");   // 2行目：初期ライト個数
        int cnt = Integer.parseInt(sc.nextLine());      // 3行目：調査回数

        // 調査回数分ループ
        for (int i = 0; i < cnt; i++) {
            int lightSum = 0;
            line = sc.nextLine().split(" ");                // 4行目以降：調査区間
            int kukanFrom = Integer.parseInt(line[0]) - 1;  // 4行目：調査区間From
            int kukanTo = Integer.parseInt(line[1]) - 1;    // 4行目：調査区間To
            int kukan = kukanTo - kukanFrom + 1;            // 調査区間数

            // 調査区間数分ループしてライトの合計値を求める
            for (int j = 0; j < kukan; j++) {
                if (j == 0) {
                    lightSum = Integer.parseInt(nowLight[kukanFrom + j]);
                } else {
                    lightSum = lightSum + Integer.parseInt(nowLight[kukanFrom + j]);
                }
            }

            // ライトの平均値が安全基準を満たしていない場合
            if (lightSum / (kukanTo - kukanFrom + 1) < light) {
                int add = (int)Math.ceil(light - lightSum / (kukanTo - kukanFrom + 1));
                for (int j = kukanFrom; j < kukan; j++) {
                    nowLight[kukanFrom] = Integer.toString(Integer.parseInt(nowLight[kukanFrom]) + add);
                    kukanFrom++;
                }
            }
        }

        // 出力
        String answer = "";
        for (int j = 0; j < nowLight.length; j++) {
            if (j == 0) {
                answer = answer + nowLight[j];
            } else {
                answer = answer + " " + nowLight[j];
            }
        }
        System.out.println(answer);
    }
}
