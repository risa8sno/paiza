import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // 1行目
        String[] temp = sc.nextLine().split(" ");
        int green = Integer.parseInt(temp[0]);  // しげみの数
        int rabbit = Integer.parseInt(temp[1]); // ウサギの数
        int jump = Integer.parseInt(temp[2]);   // ジャンプ回数

        // 2行目以降：ウサギの初期位置
        int[] rabbitPositon = new int[rabbit];
        boolean[] rabbitFlag = new boolean[green];
        for (int i = 0; i < rabbit; i++) {
            rabbitPositon[i] = Integer.parseInt(sc.nextLine()) - 1;
            rabbitFlag[rabbitPositon[i]] = true;
        }

        // ジャンプ回数分ウサギの位置を取得
        for (int i = 0; i < jump; i++) {

            // 各ウサギ分ループ
            for (int j = 0; j < rabbit; j++) {
                
                int prePosition = rabbitPositon[j];     // 移動前の位置

                // 空いている位置まで移動する
                while (rabbitFlag[rabbitPositon[j]]) {
                    rabbitPositon[j]++;
                    // 位置番号調整
                    if (rabbitPositon[j] > green - 1) {
                        rabbitPositon[j] -= green;
                    }
                }

                rabbitFlag[rabbitPositon[j]] = true;
                rabbitFlag[prePosition] = false;
            }
        }

        for (int i = 0; i < rabbit; i++) {
            System.out.println(rabbitPositon[i] + 1);
        }
    }
}
