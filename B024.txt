import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        double r = sc.nextDouble();

        int sum = 0;
        int n = (int)r;

        for (int i = 0; i <= n; i++) {
            sum += (int)Math.ceil(Math.sqrt(r*r - i*i));
        }
        System.out.println(sum * 4);
    }
}
